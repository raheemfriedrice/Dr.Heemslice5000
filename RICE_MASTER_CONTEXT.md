### Parser script

```python
# parse_anthropic_export.py
# Reads Anthropic export, updates RICE_MASTER_CONTEXT.md between marker tags only.
# Never touches the Claudit log or any hand-written section.

import json
import re
from pathlib import Path
from collections import Counter, defaultdict
from datetime import datetime

ROOT = Path(r"C:\Users\snack\Desktop\SNACKMASTER_RICE_ARCHITECTURE")
EXPORT_DIR = ROOT / "anthropic_export"
MASTER_FILE = ROOT / "RICE_MASTER_CONTEXT.md"

FRANCHISE_ANCHORS = {
    "RR-": ["rice & roni", "rice and roni", "rr-", "605 snow", "power wash", "riceandroni", "rosé", "wadsworth", "drumstick"],
    "Meyer": ["meyer", "waste-bin", "driftwood flats", "slippi", "lippi", "pickens"],
    "Dream Keeper": ["dream keeper", "solenne", "vael", "threshold"],
    "Winner Winner": ["winner winner", "cluck royale", "starter chicken", "casino-bird"],
    "Raheem Fried Rice": ["raheem fried rice", "fried rice", "heemslice", "snackcity"],
}

DECISION_MARKERS = [
    r"\bdecided\b", r"\block(ed|ing) in\b", r"\bgoing with\b", r"\bgoing to\b",
    r"\bwe'?ll\b", r"\bi'?ll\b", r"\blaunched\b", r"\bshipping\b", r"\bcommitted\b",
    r"\bfinalized\b", r"\bset on\b",
]
OPEN_LOOP_MARKERS = [
    r"\bstill need\b", r"\btodo\b", r"\bnext step\b", r"\bpending\b",
    r"\bblocker\b", r"\bunblock\b", r"\bwaiting on\b", r"\bhaven'?t\b",
]


def extract_text(message):
    content = message.get("content") or message.get("text") or ""
    if isinstance(content, list):
        parts = []
        for block in content:
            if isinstance(block, dict):
                parts.append(block.get("text", ""))
            else:
                parts.append(str(block))
        return " ".join(parts)
    return str(content)


def tag_conversation(text_lower):
    return [a for a, kws in FRANCHISE_ANCHORS.items() if any(kw in text_lower for kw in kws)]


def pull_sentences(text, patterns, max_n=3):
    sentences = re.split(r"(?<=[.!?])\s+", text)
    matches = []
    for s in sentences:
        s = s.strip()
        if 25 < len(s) < 280 and any(re.search(p, s, re.IGNORECASE) for p in patterns):
            matches.append(s)
        if len(matches) >= max_n:
            break
    return matches


def replace_between(text, start_marker, end_marker, new_content):
    pattern = re.compile(
        re.escape(start_marker) + r".*?" + re.escape(end_marker),
        re.DOTALL,
    )
    replacement = start_marker + "\n" + new_content.strip() + "\n" + end_marker
    return pattern.sub(replacement, text)


def main():
    convo_file = EXPORT_DIR / "conversations.json"
    if not convo_file.exists():
        raise FileNotFoundError(f"Not found: {convo_file}")

    with open(convo_file, "r", encoding="utf-8") as f:
        conversations = json.load(f)

    anchor_stats = Counter()
    decisions_by_anchor = defaultdict(list)
    loops_by_anchor = defaultdict(list)

    for convo in conversations:
        messages = convo.get("chat_messages", [])
        full_text = " ".join(extract_text(m) for m in messages)
        anchors = tag_conversation(full_text.lower())
        for a in anchors:
            anchor_stats[a] += 1
            decisions_by_anchor[a].extend(pull_sentences(full_text, DECISION_MARKERS, 2))
            loops_by_anchor[a].extend(pull_sentences(full_text, OPEN_LOOP_MARKERS, 2))

    # --- Build SWOT ---
    swot = [f"*Auto-generated {datetime.now().strftime('%Y-%m-%d %H:%M')} from {len(conversations)} conversations.*", ""]
    swot.append("**Strengths**")
    for a, c in sorted(anchor_stats.items(), key=lambda x: -x[1])[:3]:
        swot.append(f"- {a} — {c} conversations of sustained development")
    swot.append(f"- {sum(len(v) for v in decisions_by_anchor.values())} concrete decisions logged across the multiverse")
    swot.append(f"- Cross-franchise threading already canonized (#9e90FC, fried-rice scent on Threshold, The Glitch)")
    swot.append("")
    swot.append("**Weaknesses**")
    swot.append(f"- {sum(len(v) for v in loops_by_anchor.values())} open loops in the pipeline — context fragmentation risk")
    cold = [a for a, c in anchor_stats.items() if c < 3]
    for a in cold:
        swot.append(f"- {a} under-developed in recent sessions (low chat volume)")
    swot.append("- Token budget pressure on Pro plan forces context economy")
    swot.append("")
    swot.append("**Opportunities**")
    swot.append("- This master file eliminates re-establishment tax across every new chat")
    swot.append("- RR- Sept 18 launch anchors a real revenue date to creative work")
    swot.append("- Recurring monthly payout structures map cleanly onto RR- + music catalog + PTV")
    swot.append("")
    swot.append("**Threats**")
    swot.append("- Token depletion on Pro plan (2 days remaining at writing)")
    swot.append("- Context drift between Chat / Code / Cowork without master-file discipline")
    swot.append("- IP exposure when proprietary content enters AI systems")

    # --- Build decisions block ---
    dec_lines = []
    for anchor, items in decisions_by_anchor.items():
        if not items:
            continue
        dec_lines.append(f"**{anchor}**")
        for d in items[:5]:
            dec_lines.append(f"- {d}")
        dec_lines.append("")

    # --- Build loops block ---
    loop_lines = []
    for anchor, items in loops_by_anchor.items():
        if not items:
            continue
        loop_lines.append(f"**{anchor}**")
        for l in items[:5]:
            loop_lines.append(f"- {l}")
        loop_lines.append("")

    # --- Read master, patch, write back ---
    text = MASTER_FILE.read_text(encoding="utf-8")
    text = replace_between(text, "<!-- PARSER_SWOT_START -->", "<!-- PARSER_SWOT_END -->", "\n".join(swot))
    text = replace_between(text, "<!-- PARSER_DECISIONS_START -->", "<!-- PARSER_DECISIONS_END -->", "\n".join(dec_lines) or "_No decisions parsed._")
    text = replace_between(text, "<!-- PARSER_LOOPS_START -->", "<!-- PARSER_LOOPS_END -->", "\n".join(loop_lines) or "_No open loops parsed._")

    # Update timestamp line
    text = re.sub(
        r"\*\*Last parser run:\*\*.*",
        f"**Last parser run:** {datetime.now().strftime('%Y-%m-%d %H:%M')} — {len(conversations)} conversations across {len(anchor_stats)} anchors.",
        text,
    )

    MASTER_FILE.write_text(text, encoding="utf-8")
    print(f"Updated {MASTER_FILE.name}")
    print(f"  conversations parsed: {len(conversations)}")
    print(f"  anchor distribution:  {dict(anchor_stats)}")


if __name__ == "__main__":
    main()
```

## PART C — CLAUDIT PROTOCOL

**Definition.** A Claudit is the closing block of every chat session — a compressed record of what changed during that session, formatted so it appends cleanly to Section 7 without bloating the file or duplicating prior context.

**When to fire.** End of every meaningful chat. Trigger phrase from you: *"Close with a Claudit."* The assistant produces the block, you (or Claude Code) paste it into Section 7 between the `CLAUDIT_LOG_START` and `CLAUDIT_LOG_END` markers, newest at the bottom.

### Claudit template

```markdown
### CLAUDIT — YYYY-MM-DD — [3-5 word chat topic]
**Anchor(s) touched:** RR- | Meyer | Dream Keeper | Winner Winner | RFR | cross-franchise
**New decisions:**
- [one-line decisions made this session]
**New assets created:**
- [files, drafts, art, code committed this session — with paths if applicable]
**Open loops added:**
- [unresolved items to carry forward]
**Cross-franchise threads:**
- [anything that connected two or more anchors this session]
**Next move:**
- [the single highest-value next action]
```

### Append target

Always Section 7 of this file. Claude Code can do the paste with: *"Append this Claudit block immediately above the `<!-- CLAUDIT_LOG_END -->` marker in RICE_MASTER_CONTEXT.md."*

### Why this works for the token budget

The Claudit log compounds without you re-narrating. Each new chat: you paste Section 1 + 2 + 4 (small, dense), and if the chat touches a specific anchor, you also paste the last 2-3 Claudits for that anchor from Section 7. That replaces what would otherwise be 10-30 messages of re-establishment per chat, which is where the token bleed lives.
